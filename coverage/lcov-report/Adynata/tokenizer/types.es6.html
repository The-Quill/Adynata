export class AdynataTokenizerKeyword {
    constructor(keywordType, character, type, state){
        this.Character = character
        this.Type = type
        this.State = state
        switch (keywordType){
            case AdynataTokenizerType.NON_ALPHA:
                this.KeywordType = AdynataTokenizerType.NON_ALPHA

        }
    }
}
